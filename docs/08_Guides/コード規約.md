## TypeScript

### 型定義

#### 必須: anyを避ける
```typescript
// ❌ 悪い例
function processData(data: any) {
  return data.value;
}

// ✅ 良い例
function processData(data: { value: string }) {
  return data.value;
}
```

#### 推奨: 型を明示的に記述
```typescript
// ✅ 良い例
const count: number = 0;
const name: string = 'John';
const isActive: boolean = true;

// ⚠️ 許容 (型推論が明確な場合)
const count = 0; // number と推論される
```

#### 必須: 関数の戻り値の型を明示
```typescript
// ✅ 良い例
function calculateStability(card: Card): number {
  return card.reviewCount * 10;
}

async function fetchCards(): Promise<Card[]> {
  const response = await fetch('/api/cards');
  return response.json();
}
```

### nullとundefined

#### 推奨: Optional Chaining
```typescript
// ✅ 良い例
const userName = user?.name;
const firstCard = cards?.[0];

// ❌ 悪い例
const userName = user && user.name;
```

#### 推奨: Nullish Coalescing
```typescript
// ✅ 良い例
const count = card.reviewCount ?? 0;

// ❌ 悪い例
const count = card.reviewCount || 0; // 0 が falsy なので問題
```

## React

### コンポーネント

#### 必須: 関数コンポーネントを使用
```tsx
// ✅ 良い例
export const UserProfile: React.FC = () => {
  return <div>Profile</div>;
};

// ❌ 悪い例 (クラスコンポーネント)
export class UserProfile extends React.Component {
  render() {
    return <div>Profile</div>;
  }
}
```

#### 推奨: 早期リターン
```tsx
// ✅ 良い例
export const CardList: React.FC<CardListProps> = ({ cards }) => {
  if (cards.length === 0) {
    return <div>カードがありません</div>;
  }
  
  return (
    <div>
      {cards.map(card => <CardItem key={card.id} card={card} />)}
    </div>
  );
};

// ❌ 悪い例
export const CardList: React.FC<CardListProps> = ({ cards }) => {
  return (
    <div>
      {cards.length === 0 ? (
        <div>カードがありません</div>
      ) : (
        cards.map(card => <CardItem key={card.id} card={card} />)
      )}
    </div>
  );
};
```

### Hooks

#### 必須: Hooksのルールを守る
```tsx
// ✅ 良い例
export const UserProfile: React.FC = () => {
```

## コメント

### 必須: 日本語で記述

```typescript
// ✅ 良い例
/**
 * カードの定着度を計算する
 * @param card - 対象のカード
 * @returns 定着度 (0-100)
 */
function calculateStability(card: Card): number {
  // 復習回数に基づいて定着度を計算
  const baseStability = card.reviewCount * 10;
  
  // 最大値を100に制限
  return Math.min(100, baseStability);
}

// ❌ 悪い例 (英語)
/**
 * Calculate card stability
 */
function calculateStability(card: Card): number {
  // Calculate based on review count
  return card.reviewCount * 10;
}
```

### 推奨: JSDocを使用

```typescript
/**
 * ユーザーデータを取得する
 * @param userId - ユーザーID
 * @returns ユーザーデータ
 * @throws {Error} ユーザーが見つからない場合
 */
async function getUserData(userId: string): Promise<User> {
  const user = await db.users.get(userId);
  if (!user) {
    throw new Error('ユーザーが見つかりません');
  }
  return user;
}
```

## エラーハンドリング

### 必須: try-catchを使用

```typescript
// ✅ 良い例
async function saveCard(card: Card): Promise<void> {
  try {
    await db.cards.put(card);
    notificationService.info('保存しました');
  } catch (error) {
    console.error('カードの保存に失敗:', error);
    notificationService.error('保存に失敗しました');
  }
}

// ❌ 悪い例 (エラーハンドリングなし)
async function saveCard(card: Card): Promise<void> {
  await db.cards.put(card);
}
```

### 推奨: エラーメッセージをユーザーフレンドリーに

```typescript
// ✅ 良い例
catch (error) {
  if (error.name === 'QuotaExceededError') {
    notificationService.warning('ストレージ容量が不足しています');
  } else {
    notificationService.error('保存に失敗しました');
  }
}

// ❌ 悪い例
catch (error) {
  alert(error.message); // 技術的なエラーメッセージをそのまま表示
}
```

## パフォーマンス

### 推奨: useMemo, useCallbackを活用

```tsx
// ✅ 良い例
const sortedCards = useMemo(() => {
  return cards.sort((a, b) => a.createdAt - b.createdAt);
}, [cards]);

const handleClick = useCallback(() => {
  console.log('Clicked');
}, []);

// ⚠️ 許容 (計算コストが低い場合)
const sortedCards = cards.sort((a, b) => a.createdAt - b.createdAt);
```

### 推奨: 大きなリストは仮想化

```tsx
// ✅ 良い例 (1000件以上のリスト)
import { FixedSizeList } from 'react-window';

<FixedSizeList
  height={600}
  itemCount={cards.length}
  itemSize={50}
>
  {({ index, style }) => (
    <div style={style}>
      <CardItem card={cards[index]} />
    </div>
  )}
</FixedSizeList>

// ⚠️ 許容 (100件以下のリスト)
{cards.map(card => <CardItem key={card.id} card={card} />)}
```
